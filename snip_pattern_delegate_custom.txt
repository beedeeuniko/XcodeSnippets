// Protocol definition (in header)
@protocol CustomClassDelegate <NSObject>

@required
- (void)customClass:(id)sender didCompleteWithResult:(id)result;

@optional
- (void)customClass:(id)sender didFailWithError:(NSError *)error;
- (BOOL)customClassShouldProceed:(id)sender;

@end

// Class with delegate (in header)
@interface CustomClass : NSObject

@property (nonatomic, weak) id<CustomClassDelegate> delegate;

- (void)performAction;

@end

// Implementation
@implementation CustomClass

- (void)performAction {
    // Check if delegate implements optional method
    if ([self.delegate respondsToSelector:@selector(customClassShouldProceed:)]) {
        if (![self.delegate customClassShouldProceed:self]) {
            return;
        }
    }
    
    // Do work...
    id result = @"Success";
    
    // Call required delegate method
    [self.delegate customClass:self didCompleteWithResult:result];
}

@end
