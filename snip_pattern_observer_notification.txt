	<string>// Observer pattern using NSNotificationCenter

// Define notification name
static NSString * const MyCustomNotification = @"MyCustomNotification";

// Post notification
- (void)postNotificationWithData:(id)data {
    NSDictionary *userInfo = @{@"data": data};
    [[NSNotificationCenter defaultCenter] postNotificationName:MyCustomNotification 
                                                        object:self 
                                                      userInfo:userInfo];
}

// Register observer
- (void)registerForNotifications {
    [[NSNotificationCenter defaultCenter] addObserver:self 
                                             selector:@selector(handleNotification:) 
                                                 name:MyCustomNotification 
                                               object:nil];
}

// Handle notification
- (void)handleNotification:(NSNotification *)notification {
    id data = notification.userInfo[@"data"];
    NSLog(@"Received notification with data: %@", data);
}

// Unregister observer (important!)
- (void)dealloc {
    [[NSNotificationCenter defaultCenter] removeObserver:self];
}</string>
